# stylopoly-plugin — distributable product (VST3 / AU / Standalone).
# Thin wrapper around dehli_musikk_sampler_engine (dm::ManifestPluginProcessor).

juce_add_plugin(StyloPoly
    COMPANY_NAME            "DehliMusikk"
    BUNDLE_ID               com.dehlimusikk.stylopoly
    VERSION                 1.0.0
    PLUGIN_MANUFACTURER_CODE Dhmk
    PLUGIN_CODE              Styl
    FORMATS                 VST3 AU Standalone
    PRODUCT_NAME            "StyloPoly"
    IS_SYNTH                TRUE
    NEEDS_MIDI_INPUT        TRUE
    NEEDS_MIDI_OUTPUT       FALSE
    IS_MIDI_EFFECT          FALSE
    EDITOR_WANTS_KEYBOARD_FOCUS FALSE
    COPY_PLUGIN_AFTER_BUILD TRUE
    VST3_CATEGORIES         "Instrument" "Sampler"
    AU_MAIN_TYPE            "kAudioUnitType_MusicDevice"
)

target_sources(StyloPoly PRIVATE
    source/Plugin.cpp
)

target_compile_definitions(StyloPoly PUBLIC
    JUCE_WEB_BROWSER=0
    JUCE_USE_CURL=0
    JUCE_VST3_CAN_REPLACE_VST2=0
)

target_link_libraries(StyloPoly PRIVATE
    dehli_musikk_sampler_engine
    juce::juce_audio_utils
    juce::juce_recommended_config_flags
    juce::juce_recommended_warning_flags
)

# Embedded asset bundle (generated by ds-plugin-converter into assets/). Samples ship as a
# memory-mapped disk pack by default (installed by dmse_setup_plugin below); only the
# manifest, images and IRs are embedded.
file(GLOB STYLOPOLY_ASSETS CONFIGURE_DEPENDS
    "${CMAKE_CURRENT_SOURCE_DIR}/assets/manifest/index.json"
    "${CMAKE_CURRENT_SOURCE_DIR}/assets/manifest/modes/*.json"
    "${CMAKE_CURRENT_SOURCE_DIR}/assets/manifest/partials/*.json"
    "${CMAKE_CURRENT_SOURCE_DIR}/assets/samples/*.flac"
    "${CMAKE_CURRENT_SOURCE_DIR}/assets/ir/*.flac"
    "${CMAKE_CURRENT_SOURCE_DIR}/assets/images/*.png"
    "${CMAKE_CURRENT_SOURCE_DIR}/assets/images/*.jpg"
    "${CMAKE_CURRENT_SOURCE_DIR}/assets/images/*.jpeg")

if(STYLOPOLY_ASSETS)
    juce_add_binary_data(StyloPolyAssets SOURCES ${STYLOPOLY_ASSETS})
    target_link_libraries(StyloPoly PRIVATE StyloPolyAssets)
    target_compile_definitions(StyloPoly PRIVATE STYLOPOLY_HAS_ASSETS=1)
    list(LENGTH STYLOPOLY_ASSETS STYLOPOLY_ASSET_COUNT)
    message(STATUS "StyloPoly: embedding ${STYLOPOLY_ASSET_COUNT} asset file(s) from assets/")
else()
    message(WARNING "StyloPoly: no assets in assets/ — plugin will be silent. "
                    "Generate them with the dmse_convert target first.")
endif()

# Install the disk sample pack + wire the custom standalone settings folder.
dmse_setup_plugin(StyloPoly "StyloPoly")
